Interactive Solar System

Code Explanation

The sun is represented by a central circle, and planets orbit around it. Each planet's position is calculated using the cosine and sine of an angle, which is incremented over time to create the orbital motion. 

    
// Variable to store the sun object
let sun;

// Array to store multiple planet objects
let planets = [];

// The setup function runs once when the program starts
function setup() {
  // Create a canvas of size 800x800 pixels
  createCanvas(800, 800);

  // Initialize the sun with a radius of 50, at the center of the canvas, and a yellow color
  sun = new Planet(50, 0, 0, color(255, 204, 0));

  // Initialize two planets with different properties
  planets.push(new Planet(20, 150, 0.02, color(100, 100, 255))); // Blue planet
  planets.push(new Planet(30, 300, 0.01, color(255, 100, 100))); // Red planet
}

// The draw function runs continuously in a loop
function draw() {
  // Set the background color to black
  background(0);

  // Translate the origin point to the center of the canvas
  translate(width / 2, height / 2);

  // Display the sun
  sun.show();

  // Loop through each planet in the planets array
  for (let planet of planets) {
    // Update the planet's position
    planet.orbit();
    // Display the planet
    planet.show();
  }
}

// Function to add new planets when the mouse is pressed
function mousePressed() {
  // Generate random properties for the new planet
  let r = random(10, 30); // Random radius between 10 and 30
  let d = random(100, 400); // Random distance from the sun between 100 and 400
  let s = random(0.005, 0.03); // Random speed between 0.005 and 0.03
  let c = color(random(255), random(255), random(255)); // Random color

  // Add the new planet to the planets array
  planets.push(new Planet(r, d, s, c));
}

// Class to represent a planet
class Planet {
  // Constructor to initialize the planet's properties
  constructor(r, d, s, c) {
    this.r = r; // Radius of the planet
    this.d = d; // Distance from the sun
    this.angle = random(TWO_PI); // Initial angle for the orbit
    this.speed = s; // Speed of the orbit
    this.c = c; // Color of the planet
  }

  // Method to update the planet's position
  orbit() {
    this.angle += this.speed; // Increase the angle by the speed to create orbital motion
  }

  // Method to display the planet
  show() {
    fill(this.c); // Set the fill color to the planet's color
    noStroke(); // Disable the stroke
    let x = this.d * cos(this.angle); // Calculate the x position using cosine
    let y = this.d * sin(this.angle); // Calculate the y position using sine
    ellipse(x, y, this.r * 2); // Draw the planet as a circle
  }
}
