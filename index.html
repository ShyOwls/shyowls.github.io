p5.js Interactive 2D and 3D Shapes

Copy and paste the code into your p5.js web editor to see the shapes in action.

Example 1: Ellipse


let ellipseX = 50;
let ellipseY = 50;
let ellipseXSpeed = 2;
let ellipseYSpeed = 2;

function setup() {
  createCanvas(610, 350); // Set canvas size
}

function draw() {
  background(180); // Clear the canvas every frame

  // Parameters: ellipse(x, y, width, height)
  ellipse(ellipseX, ellipseY, 100, 50); // Draw an ellipse

  // Move the ellipse
  ellipseX += ellipseXSpeed;
  ellipseY += ellipseYSpeed;

  // Reverse direction when the ellipse hits the edge
  if (ellipseX > width - 50 || ellipseX < 50) {
    ellipseXSpeed *= -1;
  }
  if (ellipseY > height - 25 || ellipseY < 25) {
    ellipseYSpeed *= -1;
  }
}

Activity: Interactive 3D Shape


let shapeColor;
let bgColor;
let x, y;

function setup() {
  createCanvas(800, 600, WEBGL); // Create a 3D canvas
  shapeColor = color(255, 0, 0); // Initial shape color
  bgColor = color(255); // Initial background color
  background(bgColor);
  x = 0;
  y = 0;
}

function draw() {
  background(bgColor); // Set the background color
  x = map(mouseX, 0, width, -200, 200); // Map mouse X to x position
  y = map(mouseY, 0, height, -200, 200); // Map mouse Y to y position
  translate(x, y); // Move shape position
  rotateX(map(mouseY, 0, height, -PI, PI)); // Rotate based on mouse Y position
  rotateY(map(mouseX, 0, width, -PI, PI)); // Rotate based on mouse X position
  fill(shapeColor); // Set the shape color
  
  // Uncomment the shape you want to display:
  //box(100); // Draw a 3D box
  //sphere(50); // Draw a 3D sphere
  cone(50, 100); // Draw a 3D cone
  //cylinder(50, 100); // Draw a 3D cylinder
  //torus(80, 20); // Draw a 3D torus
}

function mousePressed() {
  shapeColor = color(random(255), random(255), random(255)); // Change shape color on click
  bgColor = color(random(255), random(255), random(255)); // Change background color on click
}
